# Gnuplot script for plotting the nearest-neighbor histogram output of MEAMfit
gnuplot
set terminal svg enhanced size 1929,1080 font 'Verdana,32'
set style data histograms
set style histogram rowstacked
set xtics rotate by 45 right
#set xtics font ",24"
set xtics autofreq 0.01
set key left; set key font ",32"
set xlabel 'Atomic separation (Ã…)'; set ylabel 'Frequency'; set title 'Histogram of interatomic separations for the TiN/Al203 system'
set out 'sepnHistogram.svg'
